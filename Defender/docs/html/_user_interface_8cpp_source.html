304const void UserInterface::shaderTick(double deltatime)
305{
306 static Timer<double> replaceType{ 1 / 8. };
307 static bool type = false;
308 static Timer<double> replaceWilliams{ 1. };
309 static bool williamsType = false;
310
311 if (replaceType.tick(deltatime))
312 {
313 type = !type;
314
315 if (type)
316 flashingShader->setUniform("replaceColor",
317 brightColors[rand() % 8]);
318 else
319 flashingShader->setUniform("replaceColor", sf::Glsl::Vec3(0, 0, 0));
320 }
321
322 if (replaceWilliams.tick(deltatime))
323 {
324 williamsType = !williamsType;
325
326 williamsShader->setUniform("replaceColor", williamsType ? sf::Glsl::Vec3(1, 1, 0) : sf::Glsl::Vec3(1, 0, 0));
327 }
328
329 shiftReplacement.shift(6);
330
331 shiftingShader->setUniform("replaceColor", shiftReplacement);
332
333 if (deathReplacement.active)
334 {
335 //std::cout << "ACTIVE DEATH ANIM!";
336 deathShader->setUniform("replaceColor", deathReplacement.color);
337 // If not passed white, tick
338 // if passed white
339 // if not passed hue, shift
340 // if passed, set passed to true
341 // if v == 0, set active to false
342 // else, -= 1*deltatime
343
344 if (!deathReplacement.passedWhite)
345 deathReplacement.passedWhite = deathReplacement.whiteTime.tick(deltatime);
346 else
347 {
348 if (!deathReplacement.passedHue)
349 {
350 deathReplacement.color.shift(1);
351
352 if (HSV::hasPassedHue(HSV(60, 1., 1.), deathReplacement.color, 0))
353 deathReplacement.passedHue = true;
354 }
355 else
356 {
357 deathReplacement.color.setS(1);
358 if (deathReplacement.color.getV() <= 0)
359 {
360 // Reset
361 deathReplacement.color = HSV(60, 0, 1.);
362 deathReplacement.active = false;
363 deathReplacement.passedHue = false;
364 deathReplacement.passedWhite = false;
365 }
366 else
367 deathReplacement.color.dim(1. * deltatime);
368 }
369 }
370 }
