178void Entity::applyFriction(double deltatime, double time)
179{
180 if (time == 0)
181 throw std::runtime_error("DIVISION BY 0 FRICTION");
182
183 sf::Vector2f fric = getEVel(ID);
184 fric.x /= time;
185 fric.y /= time;
186
187 fric.x *= deltatime;
188 fric.y *= deltatime;
189
190 fric.x = std::min(abs(vel.x), fric.x);
191 fric.y = std::min(abs(vel.y), fric.y);
192
193 if (vel.x < 0)
194 fric.x = -fric.x;
195 if (vel.y < 0)
196 fric.y = -fric.y;
197
198 vel -= fric;
199}

326const Entity::EntityTarget Entity::makeTargetedVec(sf::Vector2f pos, EntityID::ID ID, Entity* other,
327 uint8_t scale, bool velType)
328{
329 static constexpr float half = COMN::worldSize / 2;
330
331 sf::Vector2f target = makeCenteredTL({ 0, 0 }, other->getID());
332 sf::Vector2f eVel = getEVel(ID);
333
334 if (velType)
335 target += other->getVel();
336
337 pos = makeCenteredTL(pos, ID);
338 // Use the center & normalize
339 pos.x -= half;
340
341 // Normalize the target
342 target.x += other->getPos().x - pos.x - half;
343 target.y += other->getPos().y - pos.y;
344
345 // Apply wrapping
346 if (target.x > half)
347 target.x -= COMN::worldSize;
348 else if (target.x < -half)
349 target.x += COMN::worldSize;
350
351 // Generate the angle
352 double rot = atan2(target.y, target.x);
353
354 // Generate the velocity
355 sf::Vector2f vel = {
356 static_cast<float>(cos(rot) * scale * eVel.x),
357 static_cast<float>(sin(rot) * scale * eVel.y)
358 };
359
360 if (!velType)
361 vel += other->getVel();
362
363
364 return { vel, target };
